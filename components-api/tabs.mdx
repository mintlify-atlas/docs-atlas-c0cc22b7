## Overview

The Tabs component provides an accessible, keyboard-navigable tab interface built on Radix UI. It includes Tabs, TabsList, TabsTrigger, and TabsContent subcomponents with two visual variants.

## Components

### Tabs

Root container component for the tab interface.

<ParamField path="variant" type="'underline' | 'solid'" default="'solid'">
  Visual style variant for the tabs
  - `underline`: Tabs with bottom border indicator
  - `solid`: Tabs with solid background for active state
</ParamField>

<ParamField path="defaultValue" type="string">
  The value of the tab that should be active by default
</ParamField>

<ParamField path="value" type="string">
  The controlled value of the active tab (use with onValueChange)
</ParamField>

<ParamField path="onValueChange" type="(value: string) => void">
  Callback when the active tab changes
</ParamField>

<ParamField path="orientation" type="'horizontal' | 'vertical'" default="'horizontal'">
  Orientation of the tabs
</ParamField>

<ParamField path="dir" type="'ltr' | 'rtl'">
  Reading direction for the tabs
</ParamField>

<ParamField path="activationMode" type="'automatic' | 'manual'" default="'automatic'">
  Whether tabs activate automatically on focus or require selection
</ParamField>

### TabsList

Container for tab triggers.

<ParamField path="className" type="string">
  Additional CSS classes to apply to the list
</ParamField>

<ParamField path="ref" type="React.Ref<HTMLDivElement>">
  Ref to the underlying element
</ParamField>

<ParamField path="children" type="React.ReactNode">
  TabsTrigger components
</ParamField>

### TabsTrigger

Individual tab button.

<ParamField path="value" type="string" required>
  Unique identifier for this tab
</ParamField>

<ParamField path="disabled" type="boolean">
  Whether the tab is disabled
</ParamField>

<ParamField path="className" type="string">
  Additional CSS classes to apply to the trigger
</ParamField>

<ParamField path="ref" type="React.Ref<HTMLButtonElement>">
  Ref to the underlying button element
</ParamField>

<ParamField path="children" type="React.ReactNode">
  Content to display in the tab button
</ParamField>

### TabsContent

Content panel for each tab.

<ParamField path="value" type="string" required>
  Identifier matching the associated TabsTrigger value
</ParamField>

<ParamField path="forceMount" type="boolean">
  Force mount the content (useful for animations)
</ParamField>

<ParamField path="className" type="string">
  Additional CSS classes to apply to the content
</ParamField>

<ParamField path="ref" type="React.Ref<HTMLDivElement>">
  Ref to the underlying element
</ParamField>

<ParamField path="children" type="React.ReactNode">
  Content to display when tab is active
</ParamField>

## Usage

```tsx
import { Tabs, TabsList, TabsTrigger, TabsContent } from '@repo/ui'

export function Example() {
  return (
    <Tabs defaultValue="account" variant="underline">
      <TabsList>
        <TabsTrigger value="account">Account</TabsTrigger>
        <TabsTrigger value="password">Password</TabsTrigger>
      </TabsList>
      <TabsContent value="account">
        Account settings content
      </TabsContent>
      <TabsContent value="password">
        Password settings content
      </TabsContent>
    </Tabs>
  )
}
```

## Examples

### Underline Variant

```tsx
<Tabs defaultValue="tab1" variant="underline">
  <TabsList>
    <TabsTrigger value="tab1">Tab 1</TabsTrigger>
    <TabsTrigger value="tab2">Tab 2</TabsTrigger>
    <TabsTrigger value="tab3">Tab 3</TabsTrigger>
  </TabsList>
  <TabsContent value="tab1">Content 1</TabsContent>
  <TabsContent value="tab2">Content 2</TabsContent>
  <TabsContent value="tab3">Content 3</TabsContent>
</Tabs>
```

### Solid Variant (Default)

```tsx
<Tabs defaultValue="overview" variant="solid">
  <TabsList>
    <TabsTrigger value="overview">Overview</TabsTrigger>
    <TabsTrigger value="analytics">Analytics</TabsTrigger>
    <TabsTrigger value="reports">Reports</TabsTrigger>
  </TabsList>
  <TabsContent value="overview">
    Overview dashboard
  </TabsContent>
  <TabsContent value="analytics">
    Analytics data
  </TabsContent>
  <TabsContent value="reports">
    Reports section
  </TabsContent>
</Tabs>
```

### Controlled Tabs

```tsx
function ControlledTabs() {
  const [activeTab, setActiveTab] = useState('home')
  
  return (
    <Tabs value={activeTab} onValueChange={setActiveTab}>
      <TabsList>
        <TabsTrigger value="home">Home</TabsTrigger>
        <TabsTrigger value="profile">Profile</TabsTrigger>
      </TabsList>
      <TabsContent value="home">Home content</TabsContent>
      <TabsContent value="profile">Profile content</TabsContent>
    </Tabs>
  )
}
```

### With Disabled Tab

```tsx
<Tabs defaultValue="available">
  <TabsList>
    <TabsTrigger value="available">Available</TabsTrigger>
    <TabsTrigger value="disabled" disabled>Coming Soon</TabsTrigger>
  </TabsList>
  <TabsContent value="available">
    Available features
  </TabsContent>
</Tabs>
```

## Accessibility

- Follows WAI-ARIA tabs pattern
- Keyboard navigation with arrow keys
- Automatic or manual activation modes
- Proper ARIA labels and roles
- Focus management
